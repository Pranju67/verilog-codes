Eight to three encoder
module eight_three_encoder(
    input [7:0] X,
    output [2:0] Y
);

OR_G G1(
    .A(X[4]),
    .B(X[5]),
    .C(X[6]),
    .D(X[7]),
    .Y(Y[2])
);

OR_G G2(
    .A(X[2]),
    .B(X[3]),
    .C(X[6]),
    .D(X[7]),
    .Y(Y[1])
);

OR_G G3(
    .A(X[1]),
    .B(X[3]),
    .C(X[5]),
    .D(X[7]),
    .Y(Y[0])
);

endmodule

module OR_G(
    input A, B, C, D,
    output Y
);

assign Y = A | B | C | D;

endmodule
