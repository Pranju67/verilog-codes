# 4x1 MUX
module four_one_mux(
    input [3:0]I,
    input [1:0]S,
    output Y
    );
    wire W1,W2;
    wire [3:0]P;
    assign W1 = ~S[1];
    assign W2 = ~S[0];
    assign P[0] = W1 & W2 & I[0];
    assign P[1] = W1 & S[0] & I[1];
    assign P[2] = S[1] & W2 & I[2];
    assign P[3] = S[1] & S[0] & I[3];
    assign Y = P[0]+ P[1]+ P[2]+ P[3];
endmodule
